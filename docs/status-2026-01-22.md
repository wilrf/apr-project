import pandas as pd
from src.models.xgboost_model import UpsetXGBoost
from src.models.trainer import ModelTrainer

df = pd.read_csv("data/features/train.csv", low_memory=False)
features = pd.read_csv("data/features/columns.csv")["feature"].tolist()

model = UpsetXGBoost()
trainer = ModelTrainer(model, n_folds=6)
results = trainer.cross_validate(df, features, "upset")
